Program 2: Square Function with Unittest

import unittest def square(x): return x ** 2 class TestSquareFunction(unittest.TestCase): def test_square_positive(self): self.assertEqual(square(2), 4) self.assertEqual(square(0), 0) def test_square_negative(self): self.assertEqual(square(-2), 4) if __name__ == '__main__': unittest.main()

Program 3: Mocking Example

from unittest.mock import Mock # Function to test def get_data(): return "Real data" # Mocking the get_data function mock_get_data = Mock(return_value="Mocked data") # Testing function that uses get_data def process_data(): data = get_data() return data.upper() print(process_data()) # Output: "REAL DATA" # Using the mock instead get_data = mock_get_data print(process_data()) # Output: "MOCKED DATA"

Program 4: Divide Function with Unittest

import unittest def divide(x, y): if y == 0: raise ValueError("Cannot divide by zero") return x / y class TestDivideFunction(unittest.TestCase): def test_divide_positive(self): self.assertEqual(divide(6, 2), 3) self.assertEqual(divide(5, 1), 5) def test_divide_negative(self): self.assertEqual(divide(-6, 2), -3) self.assertEqual(divide(6, 2), 3) def test_divide_by_zero(self): with self.assertRaises(ValueError) as context: divide(4, 0) self.assertEqual(str(context.exception), "Cannot divide by zero") def test_divide_zero_numerator(self): self.assertEqual(divide(0, 2), 0) if __name__ == '__main__': unittest.main()

Program 5: Factorial Function with Pytest

import pytest def factorial(n): if n == 0: return 1 return n * factorial(n - 1) @pytest.mark.parametrize("input, expected", [ (0, 1), (1, 1), (2, 2), (5, 120), (10, 3628800), ]) def test_factorial(input, expected): assert factorial(input) == expected
